Link-https://leetcode.com/problems/employees-earning-more-than-their-managers/submissions/1728451312/
solution-
SELECT EMP.name AS Employee
FROM employee AS EMP,employee AS MGR
WHERE EMP.managerId = MGR.id AND EMP.salary>MGR.salary

Link-https://leetcode.com/problems/customers-who-never-order/submissions/1728454579/
solution-
SELECT customers.name AS Customers
FROM customers
    LEFT JOIN orders 
        on customers.id = orders.customerId
GROUP BY  customers.id
HAVING COUNT(orders.Id)=0;
